	<!-- 默认装饰器，当下面的路径都不匹配时，启用该装饰器进行装饰 -->

	<mapping path="/home" decorator="/WEB-INF/decorators/main.jsp"/>
	<mapping path="/*/*" decorator="/WEB-INF/decorators/main.jsp"/>
	
	<!-- 对不同的路径，启用不同的装饰器
